## Install

```bash
npm install --save react-layouts-v2
```

## Usage

```tsx
import * as React from 'react'

import MyComponent from 'react-layouts-v2'

class Example extends React.Component {
  render () {
    return (
      <MyComponent />
    )
  }
}
```
